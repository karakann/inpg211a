#include <stdio.h>
int sNew(int n, int k);
main()
{
    int n, k;
    printf("Obliczanie symbolu Newtona \n");
    printf("n = ");
    scanf("%d", &n);
    printf("k = ");
    scanf("%d", &k);
    if(n<k)
    {
        printf("Bledne dane \n");
    }
    else
    {
        printf("Wartosc symbolu Newtona dla %d i %d to %d \n", n, k, sNew(n, k));
    }
}
int sNew(int n, int k)
{
    int pomoc;
    if(k==0 || k==n)
    {
        return 1;
    }
    else
    {
        pomoc=sNew(n-1, k-1) + sNew(n-1, k);
        return pomoc;
    }
}
